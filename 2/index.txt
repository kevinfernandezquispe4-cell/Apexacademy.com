<script>

let game = new Chess();
let board = Chessboard('board', {draggable:true, position:'start'});

let player = {name:'Jugador', elo:800};

let players = [
{name:'Jose',elo:1200},
{name:'Maria',elo:950},
{name:'Luis',elo:700}
];

function getLevel(elo){
 if(elo<800) return 'Principiante';
 if(elo<1200) return 'Intermedio';
 return 'Avanzado';
}

function updateUI(){
 elo.innerText = "ELO: "+player.elo;
 level.innerText = "Nivel: "+getLevel(player.elo);
 renderRanking();
}

function finishGame(win){
 let change = win ? 25 : -20;
 player.elo += change;
 if(player.elo<100) player.elo=100;
 updateUI();
}

function renderRanking(){
 let all = [...players, player];
 all.sort((a,b)=>b.elo-a.elo);
 ranking.innerHTML='';
 all.forEach(p=>{
 ranking.innerHTML += `<li>${p.name} - ${p.elo}</li>`;
 });
}

updateUI();

</script>
